def self.search(query)
        if query.class == String
                query = [query]
        elsif query.class == Array
                query = query
        else
                fail 'query must be one of String or Array class'
        end
        url = "http://search.labs.crossref.org/links"
        out = 
                HTTParty.post(url, 
                        :body => query.to_json, 
                        :headers => { "Content-Type" => "application/json"}
                )
        if out.code == 200
                nil
        else
                puts "ERROR #{out.code}"
        end
        tt = out['results']
        coll = []
        tt.each do |item|
                gg = item['doi']
                if gg!=nil
                        gg = gg.sub('http://dx.doi.org/', '')
                end
                coll << 
                {
                        'match'=>item['match'], 
                        'doi'=>gg,
                        'text'=>item['text']
                }
        end
coll.display
end
